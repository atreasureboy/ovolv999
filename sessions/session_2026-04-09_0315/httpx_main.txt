(function() {
    'use strict';

    // еҲӣе»әз”»еёғе®№еҷЁ
    const container = document.createElement('div');
    container.id = 'matrix-rain-container';
    document.body.appendChild(container);
    document.body.classList.add('site-has-matrix-rain');

    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    container.appendChild(canvas);

    // ж ёеҝғй…ҚзҪ®
    const config = {
        fontSize: 18,
        speed: 3, // жӯЈеёёйҖҹеәҰпјҢдёҚеҝ«дёҚж…ў
        characterChain: '0111001010010111001010100', // еӣәе®ҡеӯ—з¬Ұй“ҫпјҢй•ҝй•ҝдёҖжқЎ
        glowIntensity: 0.6,
        columns: 0,
        density: 0.7, // жҺ§еҲ¶еҜҶеәҰпјҢ0-1д№Ӣй—ҙ
        fadeInDuration: 0.3, // ж·Ўе…Ҙж—¶й—ҙжҜ”дҫӢ
        fadeOutDuration: 0.3 // ж·ЎеҮәж—¶й—ҙжҜ”дҫӢ
    };

    let drops = [];
    let animationId = null;
    let lastTime = 0;
    const fixedCharacterChain = config.characterChain.split('');

    // еҲқе§ӢеҢ–з”»еёғе’ҢйӣЁж»ҙ
    function init() {
        resizeCanvas();
        createDrops();
        if (animationId) {
            cancelAnimationFrame(animationId);
        }
        lastTime = performance.now();
        animate();
        window.addEventListener('resize', onResize);
        window.addEventListener('visibilitychange', onVisibilityChange);
    }

    // и°ғж•ҙз”»еёғе°әеҜё
    function resizeCanvas() {
        const width = window.innerWidth;
        const height = window.innerHeight;

        if (canvas.width !== width || canvas.height !== height) {
            canvas.width = width;
            canvas.height = height;
            // ж №жҚ®еҜҶеәҰи®Ўз®—еҲ—ж•°пјҢзЎ®дҝқеқҮеҢҖеҲҶеёғ
            config.columns = Math.floor((width / config.fontSize) * config.density);
        }
    }

    // еҲӣе»әйӣЁж»ҙеҜ№иұЎ
    function createDrops() {
        drops = [];
        const columnWidth = canvas.width / config.columns;
        
        for (let i = 0; i < config.columns; i++) {
            // зЎ®дҝқе·ҰеҸіеҲҶеёғеқҮеҢҖ
            const x = i * columnWidth + Math.random() * columnWidth * 0.5;
            drops.push({
                x: x,
                y: Math.random() * -canvas.height * 2, // йҡҸжңәиө·е§ӢдҪҚзҪ®пјҢдҪҶзЎ®дҝқеқҮеҢҖ
                speed: config.speed * (0.8 + Math.random() * 0.4), // иҪ»еҫ®йҖҹеәҰеҸҳеҢ–
                characters: [...fixedCharacterChain], // еӣәе®ҡеӯ—з¬Ұй“ҫеүҜжң¬
                charPositions: new Array(fixedCharacterChain.length).fill(0),
                startTime: Date.now() + Math.random() * 5000, // йҡҸжңәејҖе§Ӣж—¶й—ҙйҒҝе…ҚеҗҢжӯҘ
                currentIndex: 0,
                isActive: false
            });
        }
    }

    // и®Ўз®—еӯ—з¬ҰйҖҸжҳҺеәҰ
    function calculateAlpha(drop, charIndex, currentY) {
        const totalChars = drop.characters.length;
        const progress = charIndex / totalChars;
        
        // ж·Ўе…ҘеҢәеҹҹ
        if (progress < config.fadeInDuration) {
            return progress / config.fadeInDuration;
        }
        // ж·ЎеҮәеҢәеҹҹ
        else if (progress > (1 - config.fadeOutDuration)) {
            return (1 - progress) / config.fadeOutDuration;
        }
        // дёӯй—ҙе®Ңе…ЁжҳҫзӨә
        else {
            return 1;
        }
    }

    // дё»еҠЁз”»еҫӘзҺҜ
    function animate(currentTime = 0) {
        const deltaTime = currentTime - lastTime;
        lastTime = currentTime;

        // дҪҝз”Ёе®Ңе…ЁйҖҸжҳҺжё…йҷӨз”»еёғпјҢйҒҝе…ҚиғҢжҷҜеҸҳжҡ—
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        ctx.font = `bold ${config.fontSize}px 'Courier New', monospace`;
        ctx.textAlign = 'center';

        for (let i = 0; i < drops.length; i++) {
            const drop = drops[i];
            
            // жҝҖжҙ»жЈҖжҹҘ
            if (!drop.isActive && Date.now() > drop.startTime) {
                drop.isActive = true;
            }
            
            if (!drop.isActive) continue;

            // жӣҙж–°дҪҚзҪ®
            drop.y += drop.speed * (deltaTime / 16); // еҹәдәҺж—¶й—ҙеўһйҮҸзЎ®дҝқеҢҖйҖҹ

            // з»ҳеҲ¶еӯ—з¬Ұй“ҫ
            for (let j = 0; j < drop.characters.length; j++) {
                const charY = drop.y - j * config.fontSize;
                
                // еҸӘз»ҳеҲ¶еұҸе№•еҶ…зҡ„еӯ—з¬Ұ
                if (charY > -config.fontSize && charY < canvas.height + config.fontSize) {
                    const alpha = calculateAlpha(drop, j, charY);
                    
                    if (alpha <= 0) continue;

                    // е®Ңе…ЁдҪҝз”ЁCSSдёӯзҡ„и“қиүІж ·ејҸ
                    const blueIntensity = Math.floor(alpha * 255);
                    const color = `rgba(0, ${blueIntensity}, ${blueIntensity}, ${alpha})`;
                    
                    // и®ҫзҪ®йҳҙеҪұж•ҲжһңеҢ№й…ҚCSS
                    ctx.shadowBlur = 10 * alpha;
                    ctx.shadowColor = `rgba(0, 255, 255, ${alpha * config.glowIntensity})`;
                    ctx.fillStyle = color;
                    
                    ctx.fillText(drop.characters[j], drop.x, charY);
                }
            }

            // йҮҚзҪ®жқЎд»¶пјҡж•ҙдёӘеӯ—з¬Ұй“ҫзҰ»ејҖеұҸе№•
            if (drop.y - (drop.characters.length * config.fontSize) > canvas.height) {
                resetDrop(drop);
            }
        }

        animationId = requestAnimationFrame(animate);
    }

    // йҮҚзҪ®йӣЁж»ҙ
    function resetDrop(drop) {
        drop.y = Math.random() * -canvas.height * 2;
        drop.x = Math.random() * canvas.width; // йҡҸжңәxдҪҚзҪ®зЎ®дҝқеқҮеҢҖ
        drop.startTime = Date.now() + Math.random() * 2000; // йҡҸжңәе»¶иҝҹ
        drop.isActive = false;
        drop.speed = config.speed * (0.8 + Math.random() * 0.4);
    }

    // е“Қеә”зӘ—еҸЈеӨ§е°ҸеҸҳеҢ–
    function onResize() {
        resizeCanvas();
        createDrops();
    }

    // йЎөйқўеҸҜи§ҒжҖ§еҸҳеҢ–еӨ„зҗҶ
    function onVisibilityChange() {
        if (document.hidden) {
            if (animationId) {
                cancelAnimationFrame(animationId);
                animationId = null;
            }
        } else {
            if (!animationId) {
                lastTime = performance.now();
                animate();
            }
        }
    }

    // еҗҜеҠЁж•°жҚ®йӣЁ
    init();

})();
